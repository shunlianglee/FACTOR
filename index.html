Supabase 10秒快速建表指南

請在您的 Supabase 控制台的 SQL Editor 貼上以下語法執行，即可與本網頁完美串接：

-- 1. 建立技術知識主表
create table if not exists knowledge_base (
  id bigint generated by default as identity primary key,
  title text not null,
  content text not null,
  category text not null,
  tags text[] default '{}'::text[],
  raw_input text,
  created_at timestamp with time zone default timezone('utc'::text, now()) not null
);

-- 2. 建立多語言翻譯表
create table if not exists knowledge_translations (
  id bigint generated by default as identity primary key,
  knowledge_id bigint references knowledge_base(id) on delete cascade not null,
  language text not null,
  translated_title text not null,
  translated_content text not null,
  created_at timestamp with time zone default timezone('utc'::text, now()) not null
);

-- 3. 安全政策：如果 RLS 被強制啟用，建立所有人（包含匿名網頁）皆可公開讀寫的政策 (Policies)
drop policy if exists "Allow public insert access on knowledge_base" on knowledge_base;
drop policy if exists "Allow public read access on knowledge_base" on knowledge_base;
drop policy if exists "Allow public insert access on knowledge_translations" on knowledge_translations;
drop policy if exists "Allow public read access on knowledge_translations" on knowledge_translations;

create policy "Allow public insert access on knowledge_base" on knowledge_base for insert with check (true);
create policy "Allow public read access on knowledge_base" on knowledge_base for select using (true);

create policy "Allow public insert access on knowledge_translations" on knowledge_translations for insert with check (true);
create policy "Allow public read access on knowledge_translations" on knowledge_translations for select using (true);

-- 4. 徹底關閉 RLS (部分 Supabase 新版專案可能限制此操作，可改用語法 3 建立 Policies)
alter table knowledge_base disable row level security;
alter table knowledge_translations disable row level security;
